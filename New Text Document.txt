metadata:
  name: standard
  annotations:
    storageclass.kubernetes.io/is-default-class: "true"

  name: pv-zk-0

  capacity:
    storage: 1Gi

  local:
    path: /mnt/data/zk-0

    required:
      nodeSelectorTerms:
